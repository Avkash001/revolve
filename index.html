ONE-TIME DATABASE SETUP
Run this SQL in your Supabase SQL Editor:
CREATE TABLE tasks ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, name text NOT NULL, status text DEFAULT 'in-progress', completed boolean DEFAULT false, recurring boolean DEFAULT false, notes text DEFAULT '', sort_order integer DEFAULT 0, date text NOT NULL, archived boolean DEFAULT false, created_at timestamptz DEFAULT now() ); CREATE TABLE daily_notes ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, date text UNIQUE NOT NULL, notes text DEFAULT '', created_at timestamptz DEFAULT now() ); ALTER TABLE tasks ENABLE ROW LEVEL SECURITY; ALTER TABLE daily_notes ENABLE ROW LEVEL SECURITY; CREATE POLICY "Allow all" ON tasks FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "Allow all" ON daily_notes FOR ALL USING (true) WITH CHECK (true);
