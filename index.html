One-time database setup — paste into Supabase SQL Editor
CREATE TABLE tasks ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, name text NOT NULL, status text DEFAULT 'in-progress', completed boolean DEFAULT false, recurring boolean DEFAULT false, repeat_type text DEFAULT 'none', repeat_days text DEFAULT '', repeat_time text DEFAULT '', notes text DEFAULT '', sort_order integer DEFAULT 0, date text NOT NULL, archived boolean DEFAULT false, client_id text DEFAULT 'default', created_at timestamptz DEFAULT now() ); CREATE TABLE daily_notes ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, date text NOT NULL, client_id text NOT NULL DEFAULT 'default', notes text DEFAULT '', created_at timestamptz DEFAULT now(), UNIQUE(date, client_id) ); CREATE TABLE clients ( id text PRIMARY KEY, name text NOT NULL, sort_order integer DEFAULT 0, created_at timestamptz DEFAULT now() ); ALTER TABLE tasks ENABLE ROW LEVEL SECURITY; ALTER TABLE daily_notes ENABLE ROW LEVEL SECURITY; ALTER TABLE clients ENABLE ROW LEVEL SECURITY; CREATE POLICY "Allow all" ON tasks FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "Allow all" ON daily_notes FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "Allow all" ON clients FOR ALL USING (true) WITH CHECK (true);
